
import java.util.*;
public class Switch_Demo 
{
	public static void main(String[] args) 
	{
		int choice;
		Scanner sc = new Scanner(System.in);
		System.out.println("Please select any one :");
		System.out.println("1 : Pin Change ");
		System.out.println("2 : Withdraw ");
		System.out.println("3 : Balance Check ");
		System.out.println("4 : Deposit ");
		choice=sc.nextInt();
		switch(choice)
		{
		case 1:
			System.out.println("The process of changing pin");
			break;
		case 2:
			System.out.println("The process of withdraw ");
			break;
		case 3:
			System.out.println("The process of balance check ");
			break;
		case 4:
			System.out.println("The process of Deposit ");
			break;
		default:
			System.out.println("Invalid Choice");
		}
		sc.close();
	}

}
